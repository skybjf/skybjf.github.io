            .mix-container {
                width: 800px;
                height: 800px;
                background: linear-gradient(90deg, white 50%, black 50%);
                position: relative;
                border-radius: 50%;
                border: 1px solid black;
                animation: rotate 10s linear infinite;
                box-sizing: border-box;
            }

            .mix-container::before {
                position: absolute;
                top: 0;
                left: 50%;
                margin-left: -200px;
                width: 400px;
                height: 400px;
                content: "";
                border-radius: 50%;
                background: radial-gradient(circle, white 23%, black calc(23% + 1px));
                transform-origin: 50% 100%;
            }

            .mix-container::after {
                position: absolute;
                bottom: 0;
                left: 50%;
                margin-left: -200px; 
                width: 400px;
                height: 400px;
                content: "";
                border-radius: 50%;
                background: radial-gradient(circle, black 23%, white calc(23% + 1px));
                transform-origin: 50% 0%;
            }

            p {
                position: absolute;
                mix-blend-mode: difference;
                line-height: 800px;
                font-size: 70px;
                left: 0;
                right: 0;
                text-align: center;
                color: #fff;
                z-index: 1;
                animation: rotate2 10s linear infinite;
            }

            @keyframes rotate {
                0% {
                    transform: rotate(0deg);
                }

                100% {
                    transform: rotate(359deg);
                }
            }

            @keyframes rotate2 {
                0% {
                    transform: rotate(0deg);
                }

                100% {
                    transform: rotate(-359deg);
                }
            }
